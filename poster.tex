  \headerbox{Learning to program is hard}{name=problem,column=0}{
    \vskip .1in
    {\raggedright\large
    \begin{itemize}
    \item Many students struggle in intro programming courses
    \item Pass rates of 60\% (or worse!) are not unusual
    \item Students have difficulty learning syntax and basic concepts
    \end{itemize}
    }
  }

  \headerbox{Exercises can help}{name=exercises,column=0,below=problem}{
    \vskip .1in
    {\raggedright\large
    \begin{itemize}
    \item Introduce syntax and concepts in ``small bites''
    \item Self-test exercises to accompany reading assignments
    \item In-class assessment
    \item Optional practice for students who need it
    \end{itemize}
    }
  }

  \headerbox{Web-based systems}{name=systems,column=0,below=exercises}{
    \vskip .1in
    {\raggedright\large
    \begin{itemize}
    \item Web-based systems are nice:
      \begin{itemize}
      \item Students can access from anywhere using a web browser
      \item No software installation (except on server)
      \end{itemize}
    \item Lots of existing systems:
      \begin{itemize}
      \item Free: CodingBat, PracticeIt!, CodeWrite, CodeAssessor, others
      \item Commercial: CodeLab, MyProgrammingLab
      \end{itemize}
    \item {\it Why create another?}
    \end{itemize}
    }
  }

  \headerbox{Exercises are the important part!}{name=exercisesimportant,column=1}{
    \vskip .1in
    {\raggedright\large
    \begin{itemize}
    \item The main issue is the {\em exercises}
      \begin{itemize}
      \item Who writes them?
      \item Who is allowed to use them?
      \item Are there lots of ready-made exercises available?
      \end{itemize}
    \end{itemize}
    }
  }
